from scipy import optimize as opt

def lossFunc(p, y, x, lamb):
    y=asarray(y)
    x=asarray(x)
    
    f=poly1d(p)
    return hstack((y-f(x), asarray(p)*(lamb/2.)**0.5))

def convenience(degree, n, regular=False, lamb=1):
    ur'''
     degree,  n, 
     
    
    PARAMS
    ---------------
    degree:   
    n:        
    regular:    False
    lamb:      regular=True  lambda


    RETURNS
    ---------------
    None
    '''
    x=linspace(0, 2*pi, 100)
    u' sin(x)'
    y=sin(x)
    
    u' n '
    sx=linspace(0, 2*pi, n)
    sy=sin(sx)+randn(n)*.3 #sigma=0.3
    
    if regular==False:
        c=polyfit(sx, sy, degree)
    else:
        c, _=opt.leastsq(lossFunc, [0]*n, args=(sy, sx, lamb**.5)) #  e^-18 
    p=poly1d(c)
    yn=p(x)
    plot(x, y, 'g', label='sin(x)')
    #plot(sx, sy, 'bo', label='sin(x) with noise')
    #
    plot(sx, sy, 'o', mfc='none', mec='b', mew=2, label='sin(x) with noise') 
    plot(x, yn, 'r', lw=2, label='polynomial fit')
    legend()
